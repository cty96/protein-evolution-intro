def analyze_mutation_effects(mutation_name, site_analysis, mutation_data):
    """Analyze the structural effects of a mutation and assign scores"""
    
    effects = {
        'hydrophilicity_change': 0,
        'surface_accessibility': 0,
        'local_interactions': 0,
        'electrostatic_effects': 0,
        'overall_stability': 0
    }
    
    scores = {
        'beneficial': 0,
        'neutral': 0,
        'deleterious': 0
    }
    
    position = mutation_data['position']
    original_aa = mutation_data['original']
    mutant_aa = mutation_data['mutant']
    
    print(f"\n--- Analyzing {mutation_name} ({original_aa}{position}{mutant_aa}) ---")
    
    # Analyze hydrophilicity effects
    if 'hydrophilicity' in site_analysis and isinstance(site_analysis['hydrophilicity'], dict):
        avg_kd = site_analysis['hydrophilicity'].get('average_kd_score', 0)
        
        # Define hydrophobic vs hydrophilic amino acids
        hydrophobic_aa = ['A', 'V', 'I', 'L', 'M', 'F', 'Y', 'W']
        hydrophilic_aa = ['R', 'K', 'D', 'E', 'Q', 'N', 'H', 'S', 'T', 'C']
        
        if original_aa in hydrophobic_aa and mutant_aa in hydrophilic_aa:
            effects['hydrophilicity_change'] = -1  # Potentially disruptive
            print(f"  Hydrophilicity change: Hydrophobic -> Hydrophilic (potentially disruptive)")
        elif original_aa in hydrophilic_aa and mutant_aa in hydrophobic_aa:
            effects['hydrophilicity_change'] = -1  # Potentially disruptive
            print(f"  Hydrophilicity change: Hydrophilic -> Hydrophobic (potentially disruptive)")
        else:
            effects['hydrophilicity_change'] = 0  # Neutral
            print(f"  Hydrophilicity change: Similar character (neutral)")
    
    # Analyze surface accessibility
    if 'surface_accessibility' in site_analysis:
        accessibility = site_analysis['surface_accessibility']
        if isinstance(accessibility, (int, float)):
            if accessibility > 0.8:
                effects['surface_accessibility'] = 1  # Surface exposed, mutations more tolerated
                print(f"  Surface accessibility: {accessibility:.2f} (surface exposed, mutations tolerated)")
            elif accessibility < 0.2:
                effects['surface_accessibility'] = -1  # Buried, mutations potentially harmful
                print(f"  Surface accessibility: {accessibility:.2f} (buried, mutations potentially harmful)")
            else:
                effects['surface_accessibility'] = 0  # Partially exposed
                print(f"  Surface accessibility: {accessibility:.2f} (partially exposed)")
    
    # Analyze specific amino acid changes
    # Cysteine mutations (potential disulfide bond disruption)
    if original_aa == 'C':
        effects['local_interactions'] = -2  # Highly deleterious
        scores['deleterious'] += 2
        print(f"  Cysteine mutation: Potential disulfide bond disruption (highly deleterious)")
    
    # Proline mutations (structural rigidity changes)
    elif original_aa == 'P' or mutant_aa == 'P':
        effects['local_interactions'] = -1  # Potentially disruptive
        scores['deleterious'] += 1
        print(f"  Proline involvement: Structural rigidity change (potentially disruptive)")
    
    # Charge changes
    elif (original_aa in ['R', 'K', 'H'] and mutant_aa in ['D', 'E']) or \
         (original_aa in ['D', 'E'] and mutant_aa in ['R', 'K', 'H']):
        effects['electrostatic_effects'] = -1  # Charge reversal
        scores['deleterious'] += 1
        print(f"  Charge reversal: {original_aa} -> {mutant_aa} (potentially disruptive)")
    
    # Size changes
    elif (original_aa in ['G', 'A'] and mutant_aa in ['F', 'W', 'Y']) or \
         (original_aa in ['F', 'W', 'Y'] and mutant_aa in ['G', 'A']):
        effects['local_interactions'] = -1  # Significant size change
        scores['deleterious'] += 1
        print(f"  Significant size change: {original_aa} -> {mutant_aa} (potentially disruptive)")
    
    # Conservative changes
    elif (original_aa in ['L', 'I', 'V'] and mutant_aa in ['L', 'I', 'V']) or \
         (original_aa in ['D', 'E'] and mutant_aa in ['D', 'E']) or \
         (original_aa in ['K', 'R'] and mutant_aa in ['K', 'R']):
        effects['local_interactions'] = 1  # Conservative
        scores['beneficial'] += 1
        print(f"  Conservative change: {original_aa} -> {mutant_aa} (likely beneficial)")
    
    else:
        effects['local_interactions'] = 0  # Neutral
        scores['neutral'] += 1
        print(f"  Moderate change: {original_aa} -> {mutant_aa} (neutral)")
    
    return effects, scores
